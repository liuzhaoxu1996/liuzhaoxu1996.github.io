# 文字溢出(...)

// limitedBox用于指定最大宽高
limitedBox(
    maxWidth: 130,
    child: Text(
        _poiName(),
        maxLines: 1,
        overflow: TextOverflow.ellipsis,
        style: TextStyle(color: Colors.white, fontSize: 12)
    )
)

String _poiName() {
    return item.article.pios == null || item.article.pois.length == 0 ? '未知' : item.article.pois[0]?.poiName ?? '未知';
}
